    多行文本出现省略号

    @mixin txtline($num){
    display: -webkit-box; /*重点，不能用block等其他，将对象作为弹性伸缩盒子模型显示*/
    -webkit-box-orient: vertical; /*从上到下垂直排列子元素（设置伸缩盒子的子元素排列方式）*/
    -webkit-line-clamp: $num; /*行数，超出几行隐藏且多余的用省略号表示...*/
    line-clamp: $num;
    overflow: hidden;
}

.twoline{
    width: 100px;
    @include txtline(2);
}
